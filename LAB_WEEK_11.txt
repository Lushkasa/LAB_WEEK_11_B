1. private fun insertToStore(
    fileInfo: FileInfo, contentUri: Uri,
    contentValues: ContentValues
) {
    executor.execute {
        val insertedUri = contentResolver.insert(
            contentUri,
            contentValues
        )
        insertedUri?.let {
            val inputStream =
                contentResolver.openInputStream(fileInfo.uri)
            val outputStream =
                contentResolver.openOutputStream(insertedUri)
            IOUtils.copy(inputStream, outputStream)
        }
    }
}
